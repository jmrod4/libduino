#include <Arduino.h>

#include <DelayedWrite.h>
#include <NiceDelay.h>

/**** main **********************************************************/


NiceDelay nice;


void setup(void)
{
  unsigned long current_time = millis();
  nice.addWrite(current_time, 13, HIGH, 0, 1000);
  nice.addWrite(current_time, 13, LOW,  200, 1000);
}


void loop(void)
{
  // blinks 10 seconds
  nice.delay(10000U);
  // then stops blinking
  nice.remove(13);
}
